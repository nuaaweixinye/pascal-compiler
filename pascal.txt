program comprehensive;
const TEN:=10, TWO:=2;
var x,y,z,ans;

  procedure sum(a,b);
  var tmp;
  begin
    tmp := a + b;
    ans := tmp
  end;

  procedure doubleandwrite(n);
  begin
    n := n * TWO;
    write(n)
  end;

  procedure nestedexample(a);
  var b;
    procedure inner(c);
    begin
      b := c + a
    end
  begin
    call inner(a);
    write(b)
  end

begin
  read(x,y);

  z := x * (y + TEN) - 5;

  if z >= 0 then
  begin
    call sum(x,z);
    write(ans)
  end
  else
    write(z);

  while x > 0 do
  begin
    x := x - 1;
    if odd x then
      write(x)
  end;

  call nestedexample(3);
  call doubleandwrite(y)
end
